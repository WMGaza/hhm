async function analyzeWithGemini(text) { if (!text.trim()) { statusMsg.innerText = "لم يتم تسجيل أي صوت."; return; } // المفتاح الخاص بك (تأكد أنه نشط في Google AI Studio) const GEMINI_API_KEY = "AlzaSyDsqmZVLSAlK1pmzhk-Qoq6w1YnqHBhmq3E"; const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${GEMINI_API_KEY}`; // البرومبت مع تعليمات صارمة للالتزام بأسلوبك (بدون فواصل أو نقاط) const prompt = `Return ONLY a JSON object for this delivery order. Keys: (customer_name, location, order_details, price). Arabic values only. NO dots, NO commas. Text: "${text}"`; try { const response = await fetch(url, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] }) }); const data = await response.json(); // فحص إذا كان هناك مشكلة في المفتاح أو الحصة المجانية if (data.error) { console.error("Gemini Error:", data.error); statusMsg.innerText = "خطأ من جوجل: " + data.error.message; return; } let responseText = data.candidates[0].content.parts[0].text; // استخراج الـ JSON من بين أي نصوص زائدة const jsonMatch = responseText.match(/\{[\s\S]*\}/); if (jsonMatch) { const cleanJson = JSON.parse(jsonMatch[0]); const row = ` ${cleanJson.customer_name || "غير معروف"} ${cleanJson.location || "غير محدد"} ${cleanJson.order_details || "بدون تفاصيل"} ${cleanJson.price || "0"} `; tableBody.innerHTML += row; statusMsg.innerText = "تم تحليل طلب حمامة ديلفري بنجاح!"; } else { throw new Error("Invalid response format"); } } catch (error) { console.error("Detailed Error:", error); statusMsg.innerText = "فشل في معالجة البيانات، يرجى المحاولة مرة أخرى."; } }
